# INTEGRITY CONSTRAINTS
## Q1)

```sql
CREATE TABLE DEPT (
  DEPT_NO NUMBER(5) PRIMARY KEY,
  DNAME VARCHAR(10),
  DLOC VARCHAR(10)
);
```

## Q2)

```sql
CREATE TABLE EMP (
  EMP_NO NUMBER(5) CONSTRAINT PK_EMP PRIMARY KEY,
  ENAME VARCHAR(10),
  JOB VARCHAR(10),
  MANAGER_NAME VARCHAR(10) DEFAULT 'Mr.K. RAM',
  HIRE_DATE DATE,
  SALARY NUMBER(30),
  COMMISSION NUMBER(30),
  DEPT_NO VARCHAR(5)
);
```

# Q3)

```sql
ALTER TABLE DEPT ADD CONSTRAINT UQ_DNAME UNIQUE (DNAME);
```

## Q4)

```sql
ALTER TABLE DEPT

```

## Q5)

```sql
ALTER TABLE EMP
ADD CONSTRAINT CHK_SALARY_RANGE CHECK (SALARY BETWEEN 10000 AND 20000);
```

## Q6)

```sql
ALTER TABLE EMP
ADD CONSTRAINT FK_DEPTNO_DEPT
FOREIGN KEY (DEPT_NO) REFERENCES DEPT(DEPT_NO);
```

## Q7)

```sql
ALTER TABLE EMP
ADD CONSTRAINT CHK_COMMISSION CHECK (COMMISSION < 0.1 * SALARY);
```
